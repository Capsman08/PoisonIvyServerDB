set foreign_key_checks=0;
create table Reports(
UID varchar(40),
UNIQUE(UID),
plant_type varchar(2),
latitude float,
longitude float,
img_location varchar(80),
UNIQUE(img_location),
FOREIGN KEY(img_location) REFERENCES Image_Locations(img_location)
);

create table Image_Locations(
UID varchar(40),
FOREIGN KEY(UID) REFERENCES Reports(UID),
img_location varchar(40),
UNIQUE(img_location)
);

create table Users 
(
UID varchar(40),
UNIQUE(UID),
FOREIGN KEY(UID) REFERENCES Reports(UID),
email varchar(40),
screenname varchar(40)
);
